CR EQU 0DH
LF EQU 0AH

DATA SEGMENT
NUMBER1 DW 5555H
NUMBER2 DW 5555H
LSBRESULT DW 0000H
MSBRESULT DW 0000H
MESSAGE1 DB ' THE 16-BIT LSB RESULT IS:'
ASCIILSBRESULT DB 4 dup(?)
 DB CR, LF, '$'
MESSAGE2 DB ' THE 16-BIT MSB RESULT IS:'
ASCIIMSBRESULT DB 4 dup(?)
DB CR, LF, '$'
DATA ENDS

CODE SEGMENT
Assume CS:Code, DS: Data
START: MOV AX, DATA
 MOV DS, AX
 MOV AX, NUMBER1
MOV BX, NUMBER2
MUL BX
MOV LSBRESULT, AX
 MOV MSBRESULT, DX
 LEA SI, ASCIILSBRESULT
 ADD SI,3
MOV AX, LSBRESULT
 AND AX, 000FH
 ORA AL, 30H
 MOV [SI], AL
DEC SI
 MOV AX, LSBRESULT 
AND AX,00F0H
 MOV CL,04H
SHR AL, CL
 MOV [SI], AL
 DEC SI
 MOV AX, LSBRESULT
AND AX, 0F00H
MOV CL,08H
 SHR AL, CL
 MOV [SI], AL
DEC SI
MOV AX, LSBRESULT
AND AX, F000H
 MOV CL,0BH
 SHR AL, CL
 MOV [SI], AL
MOV AH, 09H
 LEA DX, MESSAGE1
 INT 21H
 MOV AH,09H
LEA DX, ASCIILSBRESULT
 INT 21H
QUIT: MOV AL,0
 MOV AH,04CH
INT 21H
 CODE ENDS
 END START

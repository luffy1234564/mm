CR EQU 0DH
LF EQU 0AH
DATA SEGMENT
TABLE DB '0123456789'
HEXADECIMAL DW 000FFH
MESSAGE DB ' THE DECIMAL NUMBER IS:'
DECIMAL DB 4 dup(' ')
 DB CR, LF, '$'
DATA ENDS
CODE SEGMENT
 Assume CS:Code, DS: Data
START: MOV AX, DATA
 MOV DS, AX
MOV AX, HEXADECIMAL
MOV CX, 0000AH
 LEA BX, TABLE
 LEA SI, DECIMAL
ADD SI,3
TOP: CMP AX, CX
 JB BOTTOM
 XOR DX,DX 
DIV CX
 MOV AL,DL
 XLAT
MOV [SI], AL
 DEC SI
 JMP TOP
BOTTOM: XLAT
MOV [SI], AL
MOV AH,09H
 LEA DX,MESSAGE
 INT 21H
QUIT: MOV AL,0
MOV AH,04CH
 INT 21H 
CODE ENDS
 END START
